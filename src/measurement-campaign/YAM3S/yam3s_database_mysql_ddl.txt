CREATE TABLE `record_series` (
  `id` int unsigned NOT NULL AUTO_INCREMENT,
  `label` varchar(100) DEFAULT 'Unkown',
  `starting_up_time_in_ms_rx` int unsigned DEFAULT NULL,
  `starting_controller_unix_time_in_ms_rx` bigint unsigned DEFAULT NULL,
  `starting_up_time_in_ms_tx` int unsigned DEFAULT NULL,
  `starting_controller_unix_time_in_ms_tx` bigint unsigned DEFAULT NULL,
  `lat_in_deg_xe7_tx` int DEFAULT NULL,
  `lon_in_deg_xe7_tx` int DEFAULT NULL,
  `alt_in_mm_mean_sea_level_tx` int DEFAULT NULL,
  `starting_lat_in_deg_xe7_rx` int DEFAULT NULL,
  `starting_lon_in_deg_xe7_rx` int DEFAULT NULL,
  `starting_alt_in_mm_mean_sea_level_rx` int DEFAULT NULL,
  PRIMARY KEY (`id`),
  UNIQUE KEY `id_UNIQUE` (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci COMMENT='Recording activities.';

CREATE TABLE `rx_imu` (
  `id` int unsigned NOT NULL AUTO_INCREMENT,
  `record_series_id` int unsigned NOT NULL,
  `controller_unix_time_in_ms` bigint unsigned DEFAULT NULL,
  `up_time_in_ms` int unsigned DEFAULT NULL,
  `quat_real` float DEFAULT NULL,
  `quat_i` float DEFAULT NULL,
  `quat_j` float DEFAULT NULL,
  `quat_k` float DEFAULT NULL,
  `quat_radian_accuracy` float DEFAULT NULL,
  `mag_x` float DEFAULT NULL,
  `mag_y` float DEFAULT NULL,
  `mag_z` float DEFAULT NULL,
  `mag_accuracy` char(1) DEFAULT NULL,
  PRIMARY KEY (`id`),
  UNIQUE KEY `id_UNIQUE` (`id`),
  KEY `record_series_id` (`record_series_id`),
  CONSTRAINT `rx_imu_ibfk_1` FOREIGN KEY (`record_series_id`) REFERENCES `record_series` (`id`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci COMMENT='IMU data from the RX.';
